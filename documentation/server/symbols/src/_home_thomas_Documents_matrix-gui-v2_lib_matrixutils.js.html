 14 /**
 15 * Chacks to see if the required matrix folders exist. If not, it creates them.
 16 * @see configs.exports.cacheDir
 17 * @see configs.exports.appOutputDir
 18 * @see configs.exports.appLockDir
 19 * @memberOf mUtils
 20 */
 21 exports.checkDirs = function(){
 22 	var dirs = [];
 23 	dirs.push(configs.cacheDir);
 24 	dirs.push(configs.appOutputDir);
 25 	dirs.push(configs.appLockDir);
 26 	dirs.forEach(function(dir){
 27 		path.exists(dir, function(exists){
 28 			if(!exists){
 29 				console.log("Creating" + dir);
 30 				fs.mkdir(dir, "0777", function(){});
 31 			}
 32 		});
 33 	});
 34 }

 36 /**
 37 * Parses properly formatted data according to the freedesktop.org spec into the 
 38 * internal JSON representation used by Matrix. 
 39 * @param data String data to parse
 40 * @param fpath Path to manifest file
 41 * @see http://standards.freedesktop.org/desktop-entry-spec/latest/
 42 * @memberOf mUtils
 43 */
 44 exports.parseManifestData = function(data, fpath){
 45 	if(typeof(fpath) == 'undefined') fpath = "";
 46 	var app = {
 47 			"Application" : {
 48 				"manifestPath" : fpath
 49 			}
 50 		};
 51 	var lines = data.toString().split('\n');
 52 	lines.forEach(function(line){
 53 		var idx = line.indexOf("=");
 54 		if(idx == -1){
 55 			return;
 56 		}
 57 		var key = line.substring(0, idx);
 58 		var value = line.substring(idx + 1);
 59 		switch (key.toLowerCase()){
 60 			case "type":
 61 				if(value.toLowerCase() == "directory"){
 62 					app.Application.appName = "Submenu";
 63 				} 
 64 			break;
 65 			case "name":
 66 				app.Application.title = value; 
 67 			break;
 68 			case "icon":
 69 				app.Application.iconName = value;
 70 			break;
 71 			case "exec": 
 72 				var ignored = [ "%f", "%F", "%U" , "%u", "%d", "%D", "%n", "%N", "%c", "%v", "%m"];
 73 				for(var i in ignored){
 74 					value = value.replace( ignored[i], "");
 75 				}
 76 				value = value.replace("%k", fpath);
 77 				app.Application.appName = value;
 78 			break;
 79 			case "categories":
 80 				app.Application.categories = value;
 81 			break;
 82 			case "x-matrix-categorytarget":
 83 				app.Application.contents = value;
 84 			break;
 85 			case "x-matrix-lock":
 86 				app.Application.lock = value;
 87 			break;
 88 			case "x-matrix-description":
 89 				app.Application.appDescription = value;
 90 			break;
 91 			case "x-matrix-displaypriority":
 92 				app.Application.sortPriority = value;
 93 			break;
 94 			case "comment":
 95 				app.Application.comment = value;
 96 			break;
 97 			default:
 98 			/*      console.log("Unused key/value pair.");
 99 				console.log("key: " + key);
100 				console.log("value: " + value); */
101 			break;
102 		} 
103 	});
104 	if(app.Application.hasOwnProperty("appName")){
105 		app.Application.appName = app.Application.appName.replace("%i", "--icon " + app.Application.iconName);
106 	}
107 	app.Application.path64 = new Buffer(app.Application.manifestPath).toString("base64");
108 	return app;	
109 }
110 /**
111 * Determines if an application belongs in the submenu with the specified category.
112 * Category "*" returns true for all apps, and an empty string will be true if the 
113 * application has no categories.  
114 * @param app Application to check
115 * @param submenuCats Categories (semicolon separated) of apps to accept. 
116 * @memberOf mUtils
117 */
118 exports.isInSubmenu = function(app, submenuCats){
119 	if(submenuCats.indexOf("*") != -1){
120 		return true;
121 	}
122 	if(submenuCats == "" && !app.hasOwnProperty("categories")){
123 		return true;
124 	}else if(!app.hasOwnProperty("categories")){
125 		return false;
126 	}	
127 	var appCats = app.categories.toLowerCase().split(";");
128 	var sCats = submenuCats.toLowerCase().split(";");
129 	for(var i in sCats){
130 		for(var j in appCats){
131 			if(sCats[i] == appCats[j]){
132 				return true;
133 			}
134 		}
135 	}
136 	return false;
137 }
138 /**
139 * App Sorting helper. Compares apps first by sortPriority, and then alphabetically.
140 * Apps with a lower sortPriority will appear before ones with higher or no sortPriority.
141 * @param app1 first app to check 
142 * @param app2 sceond app to check
143 * @memberOf mUtils
144 */
145 exports.compareApps = function(app1, app2){
146 	var has1 = app1.Application.hasOwnProperty("sortPriority");
147 	var has2 = app2.Application.hasOwnProperty("sortPriority");
148 	if(has1 && ! has2){
149 		return -1;
150 	}
151 	if(has2 && !has1){
152 		return 1;
153 	}
154 	if(!has1 && ! has2){
155 		var n = (app2.Application.title<app1.Application.title)-(app1.Application.title<app2.Application.title);
156 		return n; 
157 	}
158 	var p1 = app1.Application.sortPriority;
159 	var p2 = app2.Application.sortPriority; 
160 	return p1 - p2;
161 }
162 /**
163 * Recursively deletes files in configs.exports.cacheDir. When the last file is deleted, writes output to response.
164 * @param files list of files to delete
165 * @param response HTTP response object
166 * @param hasErr Whether or not an error in deletion has already occurred
167 * @see configs.exports.cacheDir
168 * @see mUtils.exports.clearCache
169 * @private
170 * @memberOf mUtils
171 */
172 var cacheClearHelper = function(files, response, hasErr){
173 	var file = path.join(configs.cacheDir, files[files.length -1]);
174 	fs.unlink(file, function(err){
175 		files.pop();
176 		if(err) hasErr = true;
177 		if(files.length <= 0){
178 			response.writeHead(302, {"Content-Type": "text/html", "Location" : configs.cacheBuilderUrl});
179 			if(hasErr){
180 				response.write("Files may not have been deleted");
181 			}else{
182 				response.write("All files deleted");
183 			}
184 			response.end();		
185 		}else{
186 			cacheClearHelper(files, response, hasErr);
187 		}
188 	});
189 }
190 /**
191 * Deletes all apps in configs.exports.cacheDir. 
192 * @param response HTTP response used for output
193 * @see configs.exports.cacheDir
194 * @see mUtils.cacheClearHelper
195 * @memberOf mUtils
196 */
197 exports.clearCache = function(response){
198 	fs.readdir(configs.cacheDir, function(err, files){
199 		if(err){
200 			response.writeHead(500, {"Content-Type": "text/html"});
201 			response.write("Error reading Dir");
202 			response.end();
203 		}
204 		console.log("Clearing Cache");
205 		cacheClearHelper(files, response, false);
206 	});	
207 }

219 //String functions
220 /**
221 * Removes leading and trailing slashes.
222 * @param str String to clean
223 * @memberOf mUtils
224 */
225 exports.cleanStr = function(str){
226 	var beginIdx = str.indexOf("/");
227 	var endIdx = str.lastIndexOf("/");
228 	beginIdx = Math.max(beginIdx, 0);
229 	endIdx = (endIdx == -1) ? str.length : endIdx;
230 	str = str.substring(beginIdx, endIdx - beginIdx);
231 	return str;
232 }
233 /**
234 * Removes whitespace from a string.
235 * @param str String to clean
236 * @memberOf mUtils
237 */
238 exports.trim = function(str){
239 	var	str = str.replace(/^\s\s*/, ''),
240 		ws = /\s/,
241 		i = str.length;
242 	while (ws.test(str.charAt(--i)));
243 	return str.slice(0, i + 1);
244 }
245 /**
246 * Escapes unsafe HTML characters in text. Replaces &<>"' with their equivalents.
247 * @param unsafe string to clean
248 * @memberOf mUtils
249 */
250 exports.escapeHtml = function(unsafe) {
251   return unsafe
252       .replace(/&/g, "&")
253       .replace(/</g, "<")
254       .replace(/>/g, ">")
255       .replace(/"/g, """)
256       .replace(/'/g, "'");
257 }
258 /**
259 * Generates a random alphanumeric (and _ ) string of the specified length.
260 * @param len Length of string to generate
261 * @memberOf mUtils
262 */
263 exports.randomString = function(len){
264 	var chars = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXTZabcdefghiklmnopqrstuvwxyz_";
265 	var randomstring = '';
266 	for (var i=0; i<len; i++) {
267 		var rnum = Math.floor(Math.random() * chars.length);
268 		randomstring += chars.substring(rnum,rnum+1);
269 	}
270 	return randomstring;
271 }

293 /**
294 * Compares whether or not two paths are the same, ignoring leading and trailing slashes. 
295 * Case insensitive. 
296 * @param path1 first path to check
297 * @param path2 second path to check
298 * @memberOf mUtils
299 */
300 exports.pathCompare = function(path1, path2){
301 	//drop leading and trailing slashes on both
302 	path1 = mUtils.cleanStr(path1);
303 	path2 = mUtils.cleanStr(path2);
304 	
305 	//both to lower case; used for case insensitive comparison
306 	path1 = path1.toLowerCase();
307 	path2 = path2.toLowerCase();
308 	return path1 == path2;
309 }
310 /**
311 * Checks to see if uri has a given prefix.
312 * @param uri Uri to check
313 * @param prefix Prefix for which to look
314 * @memberOf mUtils
315 */
316 exports.checkPrefix = function(uri, prefix){
317 	var check = uri.substring(0, prefix.length);
318 	if( check === prefix){
319 		return true;
320 	}
321 	return false;
322 }
