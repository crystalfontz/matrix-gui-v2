1 var sys = require("sys"),
  2     http = require("http"),
  3     url = require("url"),
  4     path = require("path"),
  5     fs = require("fs"),
  6     events = require("events"),
  7     loader = require("./lib/load_static_file"),
  8     applist = require("./lib/applist"),
  9     apps = require("./lib/apps"),
 10     mUtils = require("./lib/matrixutils.js"),
 11     pageBuilder = require("./lib/pageBuilder.js"),
 12     configs = require("./lib/configs.js");;
 13 
 14 //check to make sure the required folders exist
 15 mUtils.checkDirs();
 16 /**
 17 * This is the core of the server portion of matrix. It binds to port 8080 and 
 18 * listens for requests. When a request is recieved, it determines the type
 19 * of request and calls the appropriate handler. Valid types of request are:
 20 * app list, menu page (including a submenu), app description page, application
 21 * icon, clear disk cache, launch application, application output, application 
 22 * status, and app description list (all app description pages). If none of 
 23 * those are matched, it will treate the request as a request for a static file
 24 * with web root of the process' working directory. 
 25 * @see mUtils.exports.isAppListRequest
 26 * @see applist.exports.generateAppList
 27 * @see mUtils.exports.isAppMenuRequest
 28 * @see pageBuilder.exports.createMenuPage
 29 * @see mUtils.exports.isAppDescriptionRequest
 30 * @see pageBuilder.exports.createAppPage
 31 * @see mUtils.exports.isAppIconRequest
 32 * @see loader.exports.loadIcon
 33 * @see mUtils.exports.isCacheClearRequest
 34 * @see mUtils.exports.clearCache
 35 * @see mUtils.exports.isAppLaunchRequest
 36 * @see apps.exports.launchApp
 37 * @see mUtils.exports.isAppOutputRequest
 38 * @see mUtils.exports.isAppStatusRequest
 39 * @see apps.exports.status
 40 * @see mUtils.exports.isAllAppDescriptionsRequest
 41 * @see applist.exports.generateAppDescriptions
 42 * @see loader.exports.load_static_file
 43 * @name server
 44 * @field
 45 */
 46 var server = http.createServer(function(request, response) {  
 47 	//turn URL into URI
 48 	var uri = url.parse(request.url).pathname;
 49 	console.log("new request for uri: " + uri);
 50 	//determine the type of request, and then handle it.  
 51 	if(mUtils.isAppListRequest(uri)) {
 52 		applist.generateAppList("*", response, mUtils.writeJSON);
 53 	}else if(mUtils.isAppMenuRequest(uri)){
 54 		var submenu = uri.substring(configs.appMenuPrefix.length);
 55 		pageBuilder.createMenuPage(submenu, response);
 56 	}else if(mUtils.isAppDescriptionRequest(uri)){
 57 		pageBuilder.createAppPage(uri, response); 
 58 	}else if(mUtils.isAppIconRequest(uri)){
 59 		loader.loadIcon(uri, response);
 60 	}else if(mUtils.isCacheClearRequest(uri)){
 61 		mUtils.clearCache(response);
 62 	}else if(mUtils.isAppLaunchRequest(uri)){
 63 		apps.launchApp(uri, response);
 64 	}else if (mUtils.isAppOutputRequest(uri)){
 65 		pageBuilder.createAppOutputPage(uri, response);
 66 	}else if(mUtils.isAppStatusRequest(uri)){
 67 		apps.status(uri, response);
 68 	}else if(mUtils.isAllAppDescriptionsRequest(uri)){
 69 		applist.generateAppDescriptions(uri, response);	
 70 	}else{  
 71 		loader.load_static_file(uri, response);  
 72 	}  
 73 });
 74 server.listen(8080);  
 75 sys.puts("Server running at http://localhost:8080/");
