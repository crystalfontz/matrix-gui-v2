1 var sys = require("sys"),
  2     http = require("http"),
  3     url = require("url"),
  4     path = require("path"),
  5     fs = require("fs"),
  6     events = require("events"),
  7     loader = require("./lib/load_static_file"),
  8     applist = require("./lib/applist"),
  9     appLauncher = require("./lib/apps"),
 10     mUtils = require("./lib/matrixutils.js"),
 11     pageBuilder = require("./lib/pageBuilder.js"),
 12     configs = require("./lib/configs.js");;
 13 
 14 mUtils.checkDirs();
 15 
 16 var server = http.createServer(function(request, response) {  
 17 	var uri = url.parse(request.url).pathname;
 18 	console.log("new request for uri: " + uri);  
 19 	if(mUtils.isAppListRequest(uri)) {
 20 		applist.generateAppList("*", response, mUtils.writeJSON);
 21 	}else if(mUtils.isAppMenuRequest(uri)){
 22 		var submenu = uri.substring(configs.appMenuPrefix.length);
 23 		pageBuilder.createMenuPage(submenu, response);
 24 	}else if(mUtils.isAppDescriptionRequest(uri)){
 25 		pageBuilder.createAppPage(uri, response); 
 26 	}else if(mUtils.isAppIconRequest(uri)){
 27 		loader.loadIcon(uri, response);
 28 	}else if(mUtils.isCacheClearRequest(uri)){
 29 		mUtils.clearCache(response);
 30 	}else if(mUtils.isAppLaunchRequest(uri)){
 31 		appLauncher.launchApp(uri, response);
 32 	}else if (mUtils.isAppOutputRequest(uri)){
 33 		pageBuilder.createAppOutputPage(uri, response);
 34 	}else if(mUtils.isAppStatusRequest(uri)){
 35 		appLauncher.status(uri, response);
 36 	}else if(mUtils.isAllAppDescriptionsRequest(uri)){
 37 		applist.generateAppDescriptions(uri, response);	
 38 	}else{  
 39 		loader.load_static_file(uri, response);  
 40 	}  
 41 });
 42 server.listen(8080);  
 43 sys.puts("Server running at http://localhost:8080/");
