 16 var fs = require("fs"),
 17     sys = require("sys"),
 18     path = require("path"),
 19     configs = require("./configs.js"),
 20     mUtils = require("./matrixutils.js");
 21 /**
 22 * Holds lists of apps. Each submenu has its own list of apps, which are 
 23 * stored indexed by submenu. The structure is a 2D associative array. 
 24 * @private
 25 * @field
 26 * @memberOf applist
 27 */
 28 var appList = {};
 29 /**
 30 * Tracks status of manifests being processed. As a manifest is found in the 
 31 * file system, it is given an entry, indexed by filepath, reflecting that it
 32 * is not complete (0). When a manifest is done being processed, the entry is
 33 * changed to reflect that it is done processing (value of 1). When all entries
 34 * for a given submenu are done, the app list for that submenu has been created. 
 35 * This is necessary because of the way Node handles I/O - when looping through
 36 * reaching the last one does not mean all the files have been read when the 
 37 * last one is read.
 38 * @private 
 39 * @field
 40 * @memberOf Applist
 41 */
 42 var appListStatuses = {};
 43 /**
 44 * Holds lists of application descriptions. Each submenu has its own set 
 45 * of app description, which is stored as an array indexed by submenu
 46 * name. Thus, it is an associative 2D array. 
 47 * @private
 48 * @field 
 49 * @memberOf applist
 50 */ 
 51 var appDescriptions = {};
 52 /**
 53 * Stores the app list to the cache file. JSON.stringify is used to convert the
 54 * Array for storage. 
 55 * @see applist.appList
 56 * @see configs.exports.cacheDir
 57 * @see configs.exports.appListCacheFile 
 58 * @private
 59 * @memberOf applist
 60 */
 61 var saveAppList = function(){
 62 	var appListCache = configs.cacheDir + configs.appListCacheFile;
 63 	fs.writeFile(appListCache, JSON.stringify(appList), function(err){
 64 		if(err){
 65 			console.log("Error making app list cache");
 66 		}
 67 	});
 68 }
 69 /**
 70 * Stores the app descriptions to the cache file. Also uses JSON.stringify.
 71 * @see applist.appDescriptions
 72 * @see configs.exports.appDescriptionsCacheFile
 73 * @see configs.exports.cacheDir
 74 * @private 
 75 * @memberOf applist
 76 */
 77 var saveAppDescriptions = function(){
 78 	var appDescCache = configs.cacheDir + configs.appDescriptionsCacheFile;
 79 	fs.writeFile(appDescCache, JSON.stringify(appDescriptions), function(err){
 80 		if(err){
 81 			console.log("Error making app description cache");
 82 		}
 83 	});
 84 }
 85 /**
 86 * Checks to see if all the files are done being processed, and if so 
 87 * executes the callback function with the sorted app list, then saves
 88 * it. 
 89 * @param response HTTP response object
 90 * @param submenu The requested submenu
 91 * @param callback The function to execute with the complete app list
 92 * @see applist.appListStatuses
 93 * @see applist.saveAppList
 94 * @memberOf applist
 95 */
 96 exports.checkCompletion = function(response, submenu, callback){
 97 	for(var i in appListStatuses[submenu]){
 98 		if(appListStatuses[submenu][i] == 0){
 99 			return;
100 		}
101 	}
102 	callback(appList[submenu].sort(mUtils.compareApps), response, submenu);
103 	saveAppList();
104 }
105 /**
106 * Takes an item in the filesystem (file/directory) and analyzes it. If it is 
107 * a directory, its contents are read and passed to processItem. If it is a
108 * file, it is checked to see if it is a manifest. If it is, it is read and
109 * parsed into the applist. If it is not, it is ignored. 
110 * {@link applist.exports.checkCompletion} is run after every item. 
111 * @param filePath Path to the item to analyze
112 * @param response HTTP response object
113 * @param submenu Current submenu being generated
114 * @param callback The function to execute with the complete app list
115 * @see applist.appList
116 * @see applist.exports.checkCompletion
117 * @see applist.appListStatuses
118 * @see mUtils.exports.isManifest
119 * @see mUtils.exports.parseManifestData
120 * @memberOf applist
121 */
122 exports.processItem = function(filePath, response, submenu, callback){
123 	fs.stat(filePath, function(err, stats){
124 		if(err){
125 			appListStatuses[submenu][filePath] = 1;
126 			exports.checkCompletion(response, submenu, callback);
127 			return;	
128 		}
129 		if(stats.isDirectory()){
130 			fs.readdir(filePath, function(err, files){
131 				if(err){
132 					appListStatuses[submenu][filePath] = 1;
133 					exports.checkCompletion(response, submenu, callback);	
134 					return;
135 				}
136 				var indices = [];
137 				for(var i in files){
138 					var idx = path.join(filePath, files[i]);
139 					appListStatuses[submenu][idx] = 0;
140 					indices.push(idx);	
141 				}
142 				for(var i in indices){
143 					exports.processItem(indices[i], response, submenu, callback); 
144 				}
145 				appListStatuses[submenu][filePath] = 1;
146 			});	
147 		} else{
148 			if(mUtils.isManifest(filePath)){
149 				fs.readFile(filePath, function(err, data){
150 					if(err){
151 						appListStatuses[submenu][filePath] = 1;
152 						exports.checkCompletion(response, submenu, callback);	
153 						return;
154 					}
155 					var parsed = mUtils.parseManifestData(data, filePath);
156 					if(mUtils.isInSubmenu(parsed.Application, submenu) && appListStatuses[submenu][filePath] != 1){
157 						appList[submenu].push(parsed);
158 					}
159 					appListStatuses[submenu][filePath] = 1;
160 					exports.checkCompletion(response, submenu, callback);
161 				});
162 			}else{
163 				appListStatuses[submenu][filePath] = 1;
164 				exports.checkCompletion(response, submenu, callback);
165 			}
166 		}
167 	});
168 }
169 /**
170 * Reads and parses manifests from the filesystem. It first checks to
171 * make sure the apps dir exists and is a directory, then scans it 
172 * and hands its contents off to {@link applist.exports.processItem}. 
173 * @param submenu The current submenu being generated
174 * @param response HTTP Response object
175 * @param callback Function to execute with the generated app list
176 * @see configs.exports.appsFolder
177 * @see applist.appListStatuses
178 * @see applist.exports.processItem
179 * @memberOf applist
180 */
181 exports.readFilesystemApps = function(submenu, response, callback){
182 	var filePath = configs.appsFolder;
183 	path.exists(filePath, function(exists){
184 		if(!exists){
185 			response.writeHead(404, {"Content-Type": "text/plain"});
186 			response.write("404 Not Found\n");
187 			response.end();
188 			return;
189 		}
190 		fs.stat(filePath, function(err, stats){
191 			if(err){
192 				response.writeHead(500, {"Content-Type": "text/plain"});
193 				response.write(err + "\n");
194 				response.end();
195 				return;
196 			}
197 			if(!stats.isDirectory()){
198 				response.writeHead(500, {"Content-Type": "text/plain"});
199 				response.write("Invalid App location\n");
200 				response.end();
201 				return;
202 			}
203 			fs.readdir(filePath, function(err, files){
204 				if(err){
205 	               response.writeHead(500, {"Content-Type": "text/plain"});
206         	         response.write(err + "\n");
207                 	response.end();
208                   return;
209 				}
210 				appListStatuses[submenu] = {};
211 				appList[submenu] = [];
212 				var i = 0;
213 				for(var a in files){
214 					appListStatuses[path.join(filePath, files[a])] = 0;
215 				}	
216 				for(var idx in files){
217 					exports.processItem(path.join(filePath, files[idx]), response, submenu, callback);
218 				}
219 			});
220 		});
221 	}); 
222 }
223 /**
224 * Generates the list of applications for a given submenu. First checks 
225 * { @link applist.appList } to see if the requested menu has been 
226 * generated, and then checks the disk cache file (located at
227 * {@link configs.exports.cacheDir} { @link configs.exports.appListCacheFile}).
228 * If not, it calls { @link applist.exports.readFilesystemApps} to generate
229 * the list.
230 * @param submenu The requested submenu
231 * @param response HTTP response object    
232 * @param callback function to call with the generated app list. Should 
233 * be of the form function (list, response, submenu)
234 * @see applist.appList
235 * @see configs.exports.cacheDir
236 * @see configs.exports.appListCacheFile
237 * @see applist.exports.readFilesystemApps
238 * @memberOf applist
239 */
240 exports.generateAppList = function(submenu, response, callback){
241 	var appListCache = configs.cacheDir + configs.appListCacheFile;
242 	if(appList.hasOwnProperty(submenu)){
243 		callback(appList[submenu], response, submenu);
244 		return;
245 	}
246 	path.exists(appListCache, function(exists){
247 		if(exists){
248 			fs.readFile(appListCache, function(err, data){
249 				if(err){
250 					exports.readFilesystemApps(submenu, response, callback);
251 					return;
252 				}
253 				
254 				var list = JSON.parse(data);
255 				if(list.hasOwnProperty(submenu)){
256 					callback(list[submenu], response, submenu);
257 					appList[submenu] = list[submenu];
258 					return;
259 				}else{
260 					exports.readFilesystemApps(submenu, response, callback);
261 				}
262 			});
263 		}else{
264 			exports.readFilesystemApps(submenu, response, callback);	
265 		}
266 	});
267 }
268 /**
269 * Takes the app list and processes it to generate the list of app descriptions. 
270 * Each application is checked to see if it specifies a description page. If it 
271 * does, that description page is read, processed, and added to the set of 
272 * descriptions. If not, it is then checked for a comment. If it has a comment,
273 * that is then used as the description and put into the set. Finally, if it has
274 * neither a description nor a comment, it is given a description of "No description
275 * provided". Errors in any step will result in the appropriate error message being 
276 * used as the description. Note that this function is passed as a callback to 
277 * { @link applist.exports.generateAppList}, hence the name. 
278 * @param list generated app list for the submenu
279 * @param response HTTP Response object
280 * @param submenu Currently generated submenu       
281 * @memberOf applist
282 */
283 var appDescriptionCallback = function(list, response, submenu){
284 	appDescriptions[submenu] = {};
285 
286 	for(var i in list){
287 		var app = list[i].Application;
288 		var idx = "app_" + app.path64;
289 		if(app.hasOwnProperty("appDescription")){
290 			if(path.existsSync(app.appDescription)){
291 				appDescriptions[submenu][idx] = fs.readFileSync(app.appDescription).toString("ascii");
292 			}else{
293 				appDescriptions[submenu][idx] = "App specified a nonexistant description page";
294 			}
295 			
296 		}else if(app.hasOwnProperty("comment")){
297 			appDescriptions[submenu][idx] = app.comment;
298 		}else{
299 			appDescriptions[submenu][idx] = "No description provided";	
300 		}
301 	}
302         mUtils.writeJSON(appDescriptions[submenu], response);
303         saveAppDescriptions();
304 }
305 /**
306 * Generates the app description set for all apps. URI is currently ignored, 
307 * although in the future it could be used to generate the app descriptions
308 * for a submenu. It checks { @link applist.appDescriptions} to see if it 
309 * contains the main appdescription list, and if so immediately returns it. 
310 * It will then check disk cache (Located at {@link configs.exports.cacheDir}/
311 * {@link configs.exports.appDescriptionsCacheFile). If it has the main cache,
312 * it will then send that and store it into {@link applist.appDescriptions}. 
313 * If neither of those are the case, it will use {@link applist.exports.generateAppList}
314 * with {@link applist.exports.appDescriptionCallback} as a callback to generate
315 * and then send the set of descriptions.
316 * @param uri Request URI. Currently ignored, but may be used later for app 
317 * descriptions for only a specific submenu.
318 * @param response HTTP response object    
319 * @memberOf applist
320 */
321 exports.generateAppDescriptions = function(uri, response){
322 	var submenu = "*";
323 	var cacheFile = configs.cacheDir + configs.appDescriptionsCacheFile;
324 	if(appDescriptions.hasOwnProperty(submenu) && appDescriptions[submenu].length >= 1){
325 		mUtils.writeJSON(appDescriptions[submenu], response);
326 		return;
327 	}
328 	path.exists(cacheFile, function(exists){
329 		if(exists){
330 			fs.readFile(cacheFile, function(err, data){
331 				if(err){
332 					exports.generateAppList(submenu, response, appDescriptionCallback);
333 					return;
334 				}
335 				var list = JSON.parse(data);
336 				if(list.hasOwnProperty(submenu) && list[submenu].length >= 1){
337 					mUtils.writeJSON(list[submenu], response);
338 				}else{
339 					exports.generateAppList(submenu, response, appDescriptionCallback);
340 				}				
341 			});
342 		}else{
343 			exports.generateAppList(submenu, response, appDescriptionCallback);	
344 		}
345 	});
346 
347 }
