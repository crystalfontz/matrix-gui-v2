 11 /**
 12 * URL prefix used to tell that the request is for an app list.
 13 * The app list is a JSON-encoded list of all the files in a submenu
 14 * @memberOf configs
 15 * @field
 16 */
 17 exports.appListPrefix = "/applist/";
 18 
 19 /**
 20 * URL prefix used to tell that the request is for an app menu. The
 21 * difference between a list and a menu is that a menu is fully 
 22 * generated HTML instead of JSON. 
 23 * @memberOf configs
 24 * @field
 25 */
 26 exports.appMenuPrefix = "/menu/";
 27 
 28 /**
 29 * URL prefix used to tell that the request is for an app description 
 30 * page. The app description page is read, a menu bar is added, and it
 31 * is returned as HTML
 32 * @memberOf configs
 33 * @field
 34 */
 35 exports.appDescriptionPrefix = "/app/";
 36 
 37 /**
 38 * URL prefix used to tell that the request is for the full cache of app

 41 * @field
 42 */
 43 exports.allAppDescriptionsPrefix = "/appdescriptions/";
 44 
 45 /**
 46 * URL prefix used to tell that the request is to launch an application.
 47 * @field
 48 */

 50 
 51 /**

 59 /**

 62 */
 63 exports.iconPrefix = "/icons/";
 64 
 65 /**
 66 * URL prefix used to tell that the request is a request to clear the

 68 * @memberOf configs

 71 exports.cacheClearPrefix = "/clearcache/";
 72 
 73 /**
 74 * URL prefix used to tell that the request is for an app run status.
 75 * An app run status request gives the name of the output file and 
 76 * whether or not the app is currently executing. (JSON encoded) 
 77 * @memberOf configs
 78 * @field
 79 */
 80 exports.appStatusPrefix = "/appstatus/";
 81 
 82 /**
 83 * Valid extensions for an application icon. This is important because app 
 84 * icons may be anywhere in the system. Icon requests contain the filepath
 85 * of the file that they want, _relative_to_filesystem_root_. This means 
 86 * that an app icon request could be used to get any file in the system 
 87 * without this filter. Icon requests that do not match these extension fail
 88 * with a permission denied error. 
 89 * @memberOf configs
 90 * @field
 91 */
 92 exports.iconExtensions = [".png", ".svg", ".jpg"];
 93 
 94 /**
 95 * Extensions for which the Cache-control header will be set so that they
 96 * will be cached client side. 
 97 * @memberOf configs
 98 * @field
 99 */
100 exports.cachableExtensions = [".png", ".gif", ".svg", ".jpg", ".html", ".js", ".css"];
