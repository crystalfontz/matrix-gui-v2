 26 /**
 27 * Executes an app. Output is captured and initial status is sent in respnse
 28 * @private
 29 * @param app Application object to execute
 30 * @param lockPath Path to the application lock file. It is presumed to have
 31 * been created beforehand and is deleted after.
 32 * @param response HTTP response object. It is used to send information (in 
 33 * JSON) about the application run and output location.
 34 * @see mUtils.exports.randomString
 35 * @see mUtils.exports.getAppOutputFilename
 36 * @see apps.exports.launchApp
 37 * @memberOf apps
 38 */
 39 var executeApp = function(app, lockPath, response){
 40 	var id = mUtils.randomString(configs.appRunIdLength);
 41 	var outFileName = mUtils.getAppOutputFilename(id);
 42 	var isRunningFile = mUtils.getRunningFilename(id);
 43 	var output = fs.createWriteStream(outFileName, {'flags' : 'a'});
 44 
 45 	fs.writeFile(isRunningFile, "running", function(err){
 46 		if(err){
 47 			console.log("Could not write running status file. Refreshes may not be accurate");
 48 		}
 49 		var split = app.appName.split(" ");
 50 		var executable = split[0];
 51 		var options = [];
 52 		if(split.length > 1){
 53 			options = split.slice (1);
 54 		}
 55 		var proc = cp.spawn(executable, options);
 56 		proc.stdout.on('data', function(data){
 57 			var c = data.toString("ascii");
 58 			output.write(c);
 59 		});
 60 		proc.stderr.on('data', function(data){
 61 			var c = data.toString("ascii");
 62 			output.write(c);
 63 		});
 64 		proc.on('exit', function(code){
 65 			output.destroySoon();
 66 			fs.unlink(isRunningFile, function(err){
 67 				if(err){
 68 					console.log("Could not remove running info file.");
 69 				}
 70 				if(lockPath != ""){
 71 					path.exists(lockPath, function(exists){
 72 						if(exists){
 73 							fs.unlink(lockPath, function(err){
 74 								if(err){
 75 									console.log("Could not remove lock");
 76 								}
 77 							});
 78 						}
 79 					});
 80 				}
 81 			});	
 82 		});
 83 	});
 84 	var jsonMsg = {
 85 			"runFile" : isRunningFile,
 86 			"outFile" : outFileName,
 87 			"id" : id
 88 	};
 89 	response.writeHead(200, {"Content-Type" : "text/json" });
 90 	response.write(JSON.stringify(jsonMsg));
 91 	response.end();	
 92 }
 93 /**
 94 * Reports whether or not an app is running, and sends its current output
 95 * via JSON. 
 96 * @see mUtils.exports.getAppIdFromUri
 97 * @see mUtils.exports.getAppOutputFilename
 98 * @see mUtils.exports.getRunningFilename
 99 * @memberOf apps
100 */
101 exports.status = function(uri, response){
102 	var id = mUtils.getAppIdFromUri(uri);
103 	var outFile = mUtils.getAppOutputFilename(id);
104 	var isRunningFile = mUtils.getRunningFilename(id);
105 	fs.readFile(outFile, function(err, data){
106 		if(err){
107 			response.writeHead(500);
108 			response.write("error reading output");
109 			response.end();
110 			return;
111 		}
112 		var jsonMsg = {
113 			"output" : data.toString("ascii")
114 		};
115 		path.exists(isRunningFile, function(exists){
116 			if(exists){
117 				jsonMsg.running = true;
118 			}else{
119 				jsonMsg.running = false;
120 			}
121 			response.writeHead(200, {"Content-Type" : "text/json" });
122 			response.write(JSON.stringify(jsonMsg));
123 			response.end();
124 		});
125 	});
126 }
127 /**
128 * Launches an application. It gets the requested app from the URI, parses its manifest,
129 * checks for any lock conflicts, creates the lock file (if there is one) and launches 
130 * the app. 
131 * @param uri Url request that will be parsed for the application
132 * @param response HTTP resopnse object
133 * @see mUtils.exports.getAppPathFromId
134 * @see mUtils.exports.getAppIdFromUri
135 * @see apps.executeApp
136 * @see configs.exports.appLockDir
137 * @see configs.exports.appLockExtension
138 * @memberOf apps
139 */
140 exports.launchApp = function(uri, response){
141 	//uri is the same as used for app links in pagebuilder.js
142 	var fPath = mUtils.getAppPathFromId( mUtils.getAppIdFromUri(uri));
143 	fs.readFile(fPath, function(err, data){
144 		if(err){s
145 			response.writeHead(500);
146 			response.write("error reading manifest");
147 			response.end();
148 			return;
149 		}
150 		var app = mUtils.parseManifestData(data, fPath).Application;
151 		if(app.hasOwnProperty("lock")){
152 			var lockPath = configs.appLockDir + app.lock + configs.appLockExtension;
153 			console.log("Creating lock with path" + lockPath);
154 			path.exists(lockPath, function(exists){
155 				if(!exists){
156 					fs.writeFile(lockPath, "lock", function(err){
157 						if(err){
158 							console.log("Error creating lock file");
159 						}
160 						executeApp(app, lockPath, response);
161 					});	
162 				}else{
163 					//this should really return a 409 header in this case. However, 
164 					//due to the way in which JQuery handles JSON requests, there's 
165 					//no way to specify an explicit failure handler. So we specify 
166 					//a 200 header and instead send back an error code in the JSON. 
167 					//response.writeHead(409);
168 					var err = "An instance of an application with the same application lock ( " + app.lock + " ) is already running. Please try again when the other application has completed.";
169 					response.writeHead(200, {"Content-Type" : "text/json" });					
170 					response.write(JSON.stringify({"failure" :  err  }));
171 					response.end();
172 				}	
173 			});
174 		}else{
175 			executeApp(app, "", response);	
176 		}
177 	});
178 }
