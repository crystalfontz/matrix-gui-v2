 16 var sys = require("sys"),
 17     http = require("http"),
 18     url = require("url"),
 19     path = require("path"),
 20     fs = require("fs"),
 21     events = require("events"),
 22     configs = require("./configs.js");
 23 /**
 24 * @filename Name of file to load
 25 * @param Response HTTP response object
 26 * @memberOf loader
 27 */
 28 exports.loadFileFromRoot = function(filename, response){
 29 	path.exists(filename, function(exists) {  
 30         	if(!exists) {  
 31         		response.writeHead(404, {"Content-Type": "text/plain"});  
 32        			response.write("404 Not Found\n");  
 33             		response.end();  
 34             		return;  
 35         	}  
 36   	        fs.readFile(filename, "binary", function(err, file) {  
 37   		        if(err) {  
 38         		        response.writeHead(500, {"Content-Type": "text/plain"});  
 39         		        response.write(err + "\n");  
 40        			        response.end();  
 41                 		return;  
 42             		}  
 43   			var headers = [];
 44 			var ext = filename.substring(filename.lastIndexOf(".")).toLowerCase();
 45 			if(configs.cachableExtensions.indexOf(ext) != -1){
 46 				headers = {
 47 						"Cache-Control" : "max-age=36000"
 48 					};
 49 			} 
 50  		        response.writeHead(200, headers);  
 51             		response.write(file, "binary");  
 52             		response.end();  
 53         	});  
 54     });  
 55 }
 56 /**
 57 * Loads a file from the matrix working directory or a subdir.
 58 * @param uri File URI to load
 59 * @param response HTTP response object
 60 * @memberOf loader
 61 */
 62 exports.load_static_file = function (uri, response) {  
 63     var filename = path.join(process.cwd(), uri);  
 64     exports.loadFileFromRoot(filename, response);
 65 }
 66 /**
 67 * Loads an application icon. First checks to see if it is a path relative to the
 68 * matrix folder, and if not, looks from the root of the filesystem.
 69 * @param uri URI of the icon to load
 70 * @param response HTTP response object to use to send the image
 71 * @see loader.loadFileFromRoot
 72 * @memberOf loader
 73 */
 74 exports.loadIcon = function(uri, response){
 75 	var filePath = uri.substring(configs.iconPrefix.length -1);
 76 	var relativePath = path.join(process.cwd(), filePath);
 77 	path.exists(relativePath, function(exists){
 78 		if(!exists){
 79 			exports.loadFileFromRoot(filePath, response);
 80 		}else{
 81 			exports.loadFileFromRoot(relativePath, response);
 82 		}
 83 	});
 84 }
