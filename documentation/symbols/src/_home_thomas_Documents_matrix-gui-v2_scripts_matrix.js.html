 85 			* @param name Name of parameter whose value to find. 			
 86 			*/
 87 			var getUrlParam = function ( name ) {
 88 				name = name.replace(/[\[]/,"\\\[").replace(/[\]]/,"\\\]");
 89 				var regexS = "[\\?&]"+name+"=([^&#]*)";
 90 				var regex = new RegExp( regexS );
 91 				var results = regex.exec( window.location.href );
 92 				if( results == null ) {
 93 					return "";
 94 				}else{
 95 				    return results[1];
 96 				}
 97 			}

115 			/**
116 			* Determines whether or not an app belongs in a submenu with the given categories
117 			* @param app JSON encoded application description
118 			* @param submenuCats semicolon-separated list of categories to be displayed.
119 			* if any of an app's categories (also semicolon-separated) match any of the categories to
120 			* be displayed, returns true.
121 			*/
122 			var isInSubmenu = function(app, submenuCats){
123 				//! No categories set for the app mean show in root submenu ("") only
124 				if(submenuCats == "" && !app.hasOwnProperty("categories")){
125 					return true;
126 				}else if(!app.hasOwnProperty("categories")){
127 					return false;
128 				}
129 				//! Case insensitive	
130 				var appCats = app.categories.toLowerCase().split(";");
131 				var sCats = submenuCats.toLowerCase().split(";");
132 				for(var i in sCats){
133 					for(var j in appCats){
134 						if(sCats[i] == appCats[j]){
135 							return true;
136 						}
137 					}
138 				}
139 				return false;
140 			}

385 			* Adds swiping functionality to menus.
386 			* Note that this is done completely differently then swiping through output. 
387 			* This was a design decision based on two different types of data: menus are 
388 			* designed to be paged through and thus consist of discrete blocks of data, 
389 			* whereas application output is a contigious block of text. This adds handlers 
390 			* for left and right swiping (prevPage and nextPage, respectively) as well as 
391 			* allows an upward swipe to go to the root menu. Swiping events are triggered
392 			* by the jquery.swipe plugin. 
393 			* @see prevPage
394 			* @see nextPage
395 			* @see scrollLoc
396 			* @see buildMenu
397 			*/
398 			var addMenuSwipe = function(){
399 				scrollLoc = $('html,body').scrollTop();
400 				$("#menu").swipe(function(dx, dy) {
401 					if(!allowPageSwipe){
402 						return;
403 					}
404 					if(Math.abs(dx) > 4){
405 						allowPageSwipe = false;
406 						if(dx > 0){
407 							prevPage();	
408 						}else{
409 							nextPage();	
410 						} 
411 						setTimeout(function(){
412 							allowPageSwipe=true;
413 							}, swipeInterval
414 						);
415 					}
416 					if(Math.abs(dy) > 3.5){
417 						if(dy < 0){
418 							buildMenu("");
419 						}
420 					}				
421 				}); 	
422 			}

429 			* @see scrollLoc

432 			var scroll = function(dy){

445 			* As in addMenuSwipe, jquery.swipe provides the swipe events. This adds the
446 			* event handlers to the output so that it will vertically scroll when swiped.
447 			* Swipe intervals are limited to one every swipeInterval, to avoid duplicate

451 			* @see scroll

457 						if( now - lastSwipeTime >  swipeInterval){
458 							scroll(dy);
459 							lastSwipeTime = now;
460 						}
461 					}				
462 				}); 	
463 			}

465 			/**
466 			* App Sorting helper. Compares apps first by sortPriority, and then alphabetically.
467 			* Apps with a lower sortPriority will appear ones with higher or no sortPriority.
468 			* Note that it allows for < 1 difference in priority, while maintaining convention 
469 			* of -1/0/+1 return value for comparison.
470 			* @param a first application
471 			* @param b the second application
472 			* @see apps
473 			* @see window.onload
474 			*/
475 			var appCompare = function(a, b){
476 				var aHas = a.Application.hasOwnProperty("sortPriority");
477 				var bHas = b.Application.hasOwnProperty("sortPriority");
478 				if(aHas && !bHas){
479 					return -1;
480 				}
481 				if(bHas && !aHas){
482 					return 1;
483 				}
484 				if(aHas && bHas){
485 					var r = a.Application.sortPriority - b.Application.sortPriority;
486 					r /= Math.abs(r);
487 					if(r != 0){
488 						return r;
489 					}
490 				}
491 				var tA = a.Application.title.toLowerCase();
492 				var tB = b.Application.title.toLowerCase();
493 				if(tA == tB){
494 					return 0;
495 				}
496 				if(tA > tB){
497 					return 1;
498 				}
499 				return -1;
500 			}

614 				$("#txlogo").click(function(){return false;});
615 				appsLoaded = 0;
616 				$(apps).each(function(k, v){
617 					var app = v.Application;
618 					var cacheIndex = "app_" + app.path64;
619 					var n = apps.length;
620 					$.ajax("/app/" + app.path64,{ 
621 					"success" : function(data){
622 							var cleaned = '<div class="outputDiv">'+cleanAppPage(data) + '</div>';
